  }

  }

  }
}

  class ServiceGateway {
    - List<Service> services
    + orchestrateServices(Request request): Response // Flow control operation
  }

  abstract class Service {
    {abstract}
    + processRequest(Request request): Response // Functional requirement-business logic operation
  }
}

ChartGenerator --* CSVValidator : uses
ChartDataService --* QuotaService : updates quota
QuotaService --* UserDataService : gets user data
PaymentProcessor --* QuotaService : updates quota
ServiceGateway --* PaymentProcessor : calls payment processor
ServiceGateway --* ChartGenerator : calls chart generator
ServiceGateway --* CSVValidator : calls csv validator
